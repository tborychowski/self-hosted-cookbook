services:
  db:
    image: mariadb:10.5

    container_name: seatable
    ports:
      - "3123:80"

      - ./data:/shared
    environment:
      - DB_HOST=db
      - DB_ROOT_PASSWD=PASSWORD
      - SEATABLE_SERVER_HOSTNAME=example.seatable.com # Specifies your host name.
      - TIME_ZONE=Europe/Dublin
    depends_on:
      - db
      - memcached
      - redis
