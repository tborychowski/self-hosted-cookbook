## docker-compose.yml
```yml
services:
  kanboard:
    image: kanboard/kanboard:latest
    environment:
      - PLUGIN_INSTALLER=true
    ports:
     - "3060:80"
#     - "443:443"
    volumes:
     - ./data:/var/www/app/data
     - ./plugins:/var/www/app/plugins
#     - kanboard_ssl:/etc/nginx/ssl
```
